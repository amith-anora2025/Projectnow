#include<stdio.h>
#include<string.h>
int main(){
    char temp;
    char str[]="Welcome home";
    char*ptr=&str[0];
    char*trp=&str[strlen(str)-1];
    while(ptr<trp){
           temp=*ptr;
           *ptr=*trp;
           *trp=temp;
           ptr++;
           trp--;
    }
    ptr=&str[0];
    while(*ptr!='\0'){
        printf("%c",*ptr);
        ptr++;
    }
}
